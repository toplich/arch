version: '3.8'

services:
  teleport:
    image: public.ecr.aws/gravitational/teleport-distroless-debug:17.4.8
    container_name: teleport
    volumes:
      - ./config:/etc/teleport
      - ./data:/var/lib/teleport
    restart: unless-stopped
    network_mode: "host"
